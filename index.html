Auto-push every entry to Google Sheets
Connect a Google Sheet once — every new expense entry will be appended automatically. Works with your personal Gmail, no OAuth required.
One-time setup (5 minutes):
① Go to sheets.google.com → create a new sheet called Riwayat
② In the sheet, click Extensions → Apps Script
③ Paste the script below → click Deploy → New deployment → Web app
④ Set "Who has access" to Anyone → copy the Web App URL
⑤ Paste the URL in the field below → Save
function doPost(e){ const data = JSON.parse(e.postData.contents); const sheet = SpreadsheetApp .getActiveSpreadsheet() .getActiveSheet(); if(sheet.getLastRow()===0){ sheet.appendRow([ 'Entry Date','Event Date','Event Name', 'Category','Supplier','Item','Txn Type', 'Invoice No','Invoice Amt','Paid Amt', 'Pay Mode','Paid By','Remark','Entered By' ]); } sheet.appendRow([ data.entry_date,data.event_date,data.event_name, data.category_l1,data.supplier_name,data.item_desc, data.txn_type,data.invoice_ref,data.invoice_amt, data.paid_amt,data.pay_mode,data.paid_by, data.remark,data.entered_by ]); return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); }
